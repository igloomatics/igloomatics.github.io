这次的作业题目比较经典，就简单堆放一下。

1.用带头节点的单链表表示链串，每个节点存放一个字符。设计一个算法求s中最长平台的长度，所谓平台是指连续相同字符。

#include <iostream>
using namespace std;

struct ListNode {
    char data;
    ListNode* next;
    ListNode(char c) : data(c), next(nullptr) {}
};

// 创建链表并计算最长平台长度
int longestPlatform(const string& str) {
    if (str.empty()) return 0;
    
    ListNode* head = new ListNode(0); // 头节点
    ListNode* tail = head;
    for (int i=0 ; i<str.length() ;i++) {
        char c = str[i];
        tail->next = new ListNode(c);
        tail = tail->next;
    }

    int maxLen = 1, currentLen = 1;
    ListNode* p = head->next->next;
    char currentChar = head->next->data;

    while (p) {
        if (p->data == currentChar) {
            currentLen++;
        } else {
            maxLen = max(maxLen, currentLen);
            currentChar = p->data;
            currentLen = 1;
        }
        p = p->next;
    }

    maxLen = max(maxLen, currentLen);

    // 释放链表内存
    while (head) {
        ListNode* temp = head;
        head = head->next;
        delete temp;
    }
    
    return maxLen;
}

int main() {
    string s;
    cin >> s;
    cout << "最长平台的长度是: " << longestPlatform(s) << endl;
    return 0;
}

2.有一个顺序串s，其字符仅由数字和小写字母组成。设计一个算法将s中所有数字字符放在前半部分，所有小写字母字符放在后半部分。并给出你所设计的算法的时间和空间复杂度。

思路：双指针，一个指针从字符串的头部开始向后遍历，另一个指针从字符串的尾部向前遍历（怎么一股子L-BFGS双循环的味道）。前面的指针用于找到小写字母字符，后面的指针用于找到数字字符，然后交换它们的位置。当两个指针相遇时，整个字符串就已经按照要求完成了重排。

#include <iostream>
#include <string>
using namespace std;

void rearrange(string &s) {
    int i = 0, j = s.length() - 1;

    while (i < j) {
        // 如果s[i]是数字，继续向右移动i
        while (i < j && isdigit(s[i])) {
            i++;
        }
        // 如果s[j]是字母，继续向左移动j
        while (i < j && isalpha(s[j])) {
            j--;
        }
        // 如果i < j，交换s[i]和s[j]
        if (i < j) {
            swap(s[i], s[j]);
            i++;
            j--;
        }
    }
}

int main() {
    string s;
    cout << "请输入字符串: ";
    cin >> s;

    rearrange(s);

    cout << "重排后的字符串: " << s << endl;

    return 0;
}

算法只利用了双指针，没有额外储存其它数据，空间复杂度为O(1)；双指针遍历，每个字符最多被访问一次，时间复杂度为O(n)。
