树顶点存储结构

1.双亲存储结构

顺序存储结构，

每个结点：[data,parent]

struct PNode{
  char data;
  int parent;
  PNode(char d,int p){
    data=d;
    parent=p;
  }
};

2.孩子链存储结构

每个结点：[data，指向所有孩子结点的指针]

struct SonNode{
  char data;
  vectir<SonNode* >sons;
  SonNode(){}
  SonNode(char d):data(d){}
};

3.长子兄弟链存储结构

每个结点：[data，指向长子的指针，指向下一个兄弟的指针]

struct EBNode{
  char data;
  EBNode* brother;
  EBNode* eson;
  EBNode():brother(NULL),eson(NULL){}
  EBNode(char d){
    data=d;
    brother=eson=NULL;
  }
};
